Per avviare i test con i DCT eseguire i seguenti comandi:
    Single-Social:
        python test.py --experiment="Single" --social="Facebook"
        python test.py --experiment="Single" --social="Instagram"
        python test.py --experiment="Single" --social="Twitter"
    
    Multi-Social:
        python test.py --experiment="Multi" 
    
    Cross-Social:
        python test.py --experiment="Cross" --social="Twitter"
        python test.py --experiment="Cross" --social="Facebook"
        python test.py --experiment="Cross" --social="Instagram"
